USE nishijuku;

-- 既にあれば削除
DROP TABLE IF EXISTS roles;

-- roles テーブル作成
create table nishijuku.`roles` (
  id int(11) not null comment '権限ID'
  , role_name varchar(20) not null comment '権限名'
  , description varchar(50) comment '権限説明'
  , primary key (id)
)
  comment='権限'
  engine InnoDB
  row_format DYNAMIC
  collate utf8mb4_general_ci
