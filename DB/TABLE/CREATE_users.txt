USE nishijuku;

-- 既にあれば削除
DROP TABLE IF EXISTS users;

create table nishijuku.`users` (
  id int(11) auto_increment not null comment 'ID'
  , login_id varchar(50) comment 'ログインID'
  , password_hash varchar(255) comment 'パスワード'
  , name varchar(50) comment '名前'
  , role_id int(11) comment '権限ID'
  , created_at datetime comment '作成日'
  , updated_at datetime comment '更新日'
  , primary key (id)
)
  comment='ユーザ'
  engine InnoDB
  row_format DYNAMIC
  auto_increment 5
  collate utf8mb4_general_ci
